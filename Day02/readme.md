book {isbn, title, language, mrp, publisherid, release_date}
author {authorid, firstName, lastName, phone, emailid} 
book_author {id, isbn, authorid} 
publisher {empno, ename, job, mgr, hiredate, sal, comm, deptno, hobbyid}
